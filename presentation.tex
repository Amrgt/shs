\begin{frame}
\section{Thèses et hypothèses}
\frametitle{Thèses et hypothèses}
\begin{itemize}
\item Les possibilités qu'offrent les Google Glass ne sont pas vraiment nouvelles mais la manière dont on utilise ces fonctionnalités est radicalement différente
\item Peur généralisée d'une fin de la vie privée si l'appareil se démocratise
\item Réaction démesurée et/ou inattendue de la société face à un appareil dont ils ne comprennent pas le fonctionnement
\item Aspects juridiques
	\begin{enumerate}
	\item Droit à l'image
	\item Conduite
	\item Propriété intellectuelle
	\end{enumerate}
\end{itemize}

\end{frame}
